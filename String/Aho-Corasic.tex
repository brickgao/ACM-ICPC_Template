int next[MAXN];

void getNext(char t[]) {
    int m = strlen(t);
    int j = 0, k = -1;
    next[0] = -1;
    while(j < m) {
        if(k == -1 || t[j] == t[k])
            next[++j] = ++k;
        else 
            k = next[k];
    }
}

int kmpCount(char s[], char t[], int c[]) {
    int ans = 0;
    int j = 0;
    int n = strlen(s);
    int m = strlen(t);
    if(n == 1 && m == 1) {
        if(s[0] == t[0])
            return 1;
        else 
            return 0;
    }
    getNext(t);
    for(int i = 0; i < n; i ++) {
        while(j > 0 && s[i] != t[j])
            j = next[j];
        if(s[i] == t[j])
            j ++;
        if(j == m) {
            c[ans++] = i - m + 1;
            j = next[j];
        }
    }
    return ans;
}
